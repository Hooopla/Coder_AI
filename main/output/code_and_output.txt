```python
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def closest_prime(num):
    if num < 2:
        return 2  # 2 is the first prime number

    # Searching upwards for the next prime
    upper = num
    while not is_prime(upper):
        upper += 1
    
    # Searching downwards for the previous prime
    lower = num
    while lower > 1 and not is_prime(lower):
        lower -= 1
        
    # Determine which prime is closer
    if abs(num - lower) <= abs(num - upper):
        return lower
    return upper

# Example user input
user_input = 10
result = closest_prime(user_input)
print(f"The closest prime number to {user_input} is {result}.")
```

Output:
```
The closest prime number to 10 is 11.
```
